~ - рабочая директория
/- корневая директория
pwd - показать дирректорию, в которой нахожусь
cd - Перейти в директорию
ls - Показать файлы и папки директории
ls -a - расшириренный список файлов
ls -l - список файлов с правами(первое это права администратора, второе группы, третье пользователя)
clear - скрыть предыдущие команды


touch, mkdir - Создание файлов и директорий
touch test.txt - создать один файл
touch test1.txt test2.txt etc. - создать несколько файлов
mkdir dir - создать папку
mkdir dir1 dir2 - создать несколько папок
mkdir -p dir1/dir-inside/dir-deeper-inside - флаг -p создаёт все папки указанного пути


cp название_файла Название_директироии куда копируем - копирует файл
mv название_файла Название_директироии куда переносим - перемещаем файл
cat - читает файл

rm - удаляет файл
rm name_folder/* - удаляет всё содержимое папки
rm -r - удаляет папку и всё содержимое
rmdir - удаляет директорию
<command> --help - показать документацию по указанной команде

Нажать два раза tab для того, чтобы подставить варианты консоли




mkdir second-project && cd second-project && ls - для одновременного вызова нескольких команд, между ними ставится апперсанд &&

sudo - перед командой это значит что она выполняется от имени администратора
sudo su - перейти в режим администратора
su <name_user> - выйти из режима администратора(switch user)

sudo chown group_name:user_name file_name - Смена группы или владельца файла
sudo chmod 664 test.txt
	4 - для чтения
	6 - для записи
	7 - полный доступ(лучше не делать) 
	
sudo chown -R group_name:user_name dir_name - Смена группы или владельца файла
sudo chmod -R 764 dir_name - Смена группы или владельца файла
	4 - для чтения
	6 - для записи
	7 - полный доступ(лучше не делать)
